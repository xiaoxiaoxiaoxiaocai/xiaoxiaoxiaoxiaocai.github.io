据曾经做过的一个题目，需要python像素旋转。

使用PIL库对图片每一行像素进行循环移位：备份每一行的像素值，计算一个随机的起始点，然后对这一行的像素进行循环移位，最后保存并显示处理后的图像。通过这种方式，图像的每一行都会有一个随机的循环移位效果。

from PIL import Image
from random import randint


f = "C:\\Users\\86157\\Desktop\\新建文件夹4\\新建文件夹 (2)\\1123.png"
img = Image.open(f)


pixels = img.load()

for i in range(img.size[0]):
    back_row = []
    for j in range(img.size[1]):
        back_row.append(pixels[i, j])

    start = randint(0, img.size[1])  
    for j in range(img.size[1]):
        pixels[i, (j + start) % img.size[1]] = back_row[j]

img.show()
img.save("C:\\Users\\86157\\Desktop\\新建文件夹4\\新建文件夹 (2)\\f11.png")

from PIL import Image  # 导入处理图片的库

# 定义一个函数，用于根据指定位置对像素行进行循环移位
def rotate_pixels(pixels, pivot):
    """根据pivot点旋转像素"""
    return pixels[pivot:] + pixels[:pivot]

# 图片路径
image_path = 'f1123.png'

# 打开图片并转换为RGB模式，获取图片的宽度和高度
p = Image.open(image_path).convert('RGB')
width, height = p.size

# 读取图片的所有像素到一个二维列表中
pixels = [[p.getpixel((x, y)) for y in range(height)] for x in range(width)]

# 初始化一个列表，用于存储每行颜色变化的像素索引
change_indices = []

# 遍历每行像素，寻找首个红色像素后颜色变化的位置
for pixel_line in pixels:
    found_red = False  # 标记是否找到过红色像素
    # 从第二像素开始遍历，避免初始的索引越界
    for idx, pixel in enumerate(pixel_line[1:], start=1):
        if found_red and pixel != pixel_line[idx - 1]:  # 找到红后颜色变化
            change_indices.append(idx)  # 记录变化位置
            break
        elif pixel == (255, 0, 0):  # 找到红色像素
            found_red = True
    else:  # 如果一行内没发现变化，记录None
        change_indices.append(None)

# 根据找到的索引进行像素行的旋转处理
rotated_pixels = []
for idx, pivot in enumerate(change_indices):
    if pivot is not None:  # 如果找到了旋转点
        rotated_pixels.append(rotate_pixels(pixels[idx], pivot))
    else:  # 没有旋转点，保持原行
        rotated_pixels.append(pixels[idx])

# 创建新图片并放置处理后的像素
new_image = Image.new('RGB', (width, height))
for x in range(width):
    for y in range(height):
        new_image.putpixel((x, y), rotated_pixels[x][y])

# 保存处理后的新图片
new_image.save('flag1123_processed.png')

from PIL import Image

def get_secret(s):
    with open(s, 'rb') as f:
        data = f.read()
    return ''.join(format(byte, '08b') for byte in data)
     
def encode(original_file, hide_string, hide_file):
    im = Image.open(original_file)
    w, h = im.size
    secret_data = get_secret(hide_string)
    index = 0

    for x in range(w):
        for y in range(h):
            if index >= len(secret_data):
                break  # 已经嵌入所有秘密数据
            pixel = list(im.getpixel((x, y)))  # 将元组转为列表以便修改
            
            # 分别在R、G、B通道的最低位嵌入数据
            for channel in range(3):
                if index < len(secret_data):  # 确保索引不超出秘密数据长度
                    pixel[channel] = (pixel[channel] & ~1) | int(secret_data[index])  # 使用位运算修改最低位
                    index += 1
            
            im.putpixel((x, y), tuple(pixel))  # 将修改后的像素放回图像

    im.save(hide_file)

original_file = '1.png'
hide_string = 'flag.txt'
hide_file = 'flag.png'
encode(original_file, hide_string, hide_file)
